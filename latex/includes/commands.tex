% "\PRLsep" - PRL style horizontal line with text
% https://tex.stackexchange.com/a/212610
\newlength{\PRLlen}
\newcommand*\PRLsep[2]{\settowidth{\PRLlen}{#1}\advance\PRLlen by
-\textwidth\divide\PRLlen by
-2\noindent\makebox[\the\PRLlen]{\resizebox{\the\PRLlen}{#2}{$\blacktriangleleft$}}\raisebox{-.5ex}{#1}\makebox[\the\PRLlen]{\resizebox{\the\PRLlen}{#2}{$\blacktriangleright$}}\bigskip}
